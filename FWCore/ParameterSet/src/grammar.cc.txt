using namespace edm;

// ----------------------------------------------------------------------
// Bool_rule

bool
  edm::grammar::Bool_rule( parser & p, std::string & s )
{
  if( p.consume(LITERAL_TRUE) )  {
    encode(s, true);
    return true;
  }

  if( p.consume(LITERAL_FALSE) )  {
    encode(s, false);
    return true;
  }

  return false;
}  // Bool_rule
